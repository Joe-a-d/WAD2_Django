201def calc(user, dog): 

202 fields = ["breed", "size", "age", "gender", "houseTrained", "energyLevel"] 

203 score = 0 

204 

205 for field in fields: 

206 try: 

207 value = getattr(user,field) 

208 if value == "Any": 

209 score += 1 

210 continue 

211 elif value == getattr(dog,field): 

212 score += 1 

213# check only not null preferences , prevents skewed averages 

214 except: 

215 score += 1 

216 continue 

217 

218 return (score/len(fields))*100
